- [ ] **Step 2: Create request endpoint with rate limiting**

`functions/api/auth/email/request.ts`:
```typescript
import type { Env } from "../../../lib/env";
import { sendMagicLink, validateReturnTo } from "../../../lib/auth";
import { sendMagicLink as sendEmail } from "../../../lib/email";

export const onRequestPost: PagesFunction<Env> = async ({ request, env }) => {
  const body = await request.json<{ email: string; return_to?: string }>();
  const email = body.email?.trim().toLowerCase();

  if (!email || !email.includes("@")) {
    return Response.json({ error: "Valid email required" }, { status: 400 });
  }

  // Rate limit by email (3/hour)
  const emailKey = `rate:email:${email}`;
  const emailCount = Number(await env.SESSIONS.get(emailKey) ?? "0");
  if (emailCount >= 3) {
    return Response.json({ message: "If an account exists, we sent a link." }); // anti-enumeration
  }
  await env.SESSIONS.put(emailKey, String(emailCount + 1), { expirationTtl: 3600 });

  // Rate limit by IP (10/hour)
  const ip = request.headers.get("CF-Connecting-IP") ?? "unknown";
  const ipKey = `rate:ip:${ip}`;
  const ipCount = Number(await env.SESSIONS.get(ipKey) ?? "0");
  if (ipCount >= 10) {
    return Response.json({ message: "If an account exists, we sent a link." });
  }
  await env.SESSIONS.put(ipKey, String(ipCount + 1), { expirationTtl: 3600 });

  // Generate token
  const token = crypto.randomUUID();
  const returnTo = validateReturnTo(body.return_to ?? null);

  await env.SESSIONS.put(
    `email-login:${token}`,
    JSON.stringify({ email, returnTo }),
    { expirationTtl: 900 }, // 15 minutes
  );

  // Send email
  await sendEmail(env, email, token);

  // Always same response (anti-enumeration)
  return Response.json({ message: "If an account exists, we sent a link." });
};
```

Note: the import path for `validateReturnTo` is `../../../lib/auth` from inside `functions/api/auth/email/`.

`functions/api/auth/email/verify.ts`:
```typescript
import type { Env } from "../../../lib/env";
import { createSession, setSessionCookie } from "../../../lib/auth";

export const onRequestGet: PagesFunction<Env> = async ({ request, env }) => {
  const url = new URL(request.url);
  const token = url.searchParams.get("token");

  if (!token) return new Response("Missing token", { status: 400 });

  const data = await env.SESSIONS.get(`email-login:${token}`, "json") as
    { email: string; returnTo: string } | null;

  if (!data) return new Response("Invalid or expired link", { status: 400 });

  // Delete token (single-use)
  await env.SESSIONS.delete(`email-login:${token}`);

  const email = data.email.trim().toLowerCase();

  // Find or create profile by email
  let profile = await env.DB.prepare(
    "SELECT id, is_admin FROM profiles WHERE LOWER(TRIM(email)) = ?",
  ).bind(email).first<{ id: number; is_admin: number }>();

  if (!profile) {
    const username = email.split("@")[0].replace(/[^a-z0-9]/g, "");
    // Handle collision
    let finalUsername = username;
    const existing = await env.DB.prepare("SELECT id FROM profiles WHERE username = ?")
      .bind(finalUsername).first();
    if (existing) {
      finalUsername = `${username}-${Math.random().toString(36).slice(2, 6)}`;
    }

    const result = await env.DB.prepare(
      `INSERT INTO profiles (name, username, email, auth_provider, email_verified, role)
       VALUES (?, ?, ?, 'email', 1, 'member')`,
    ).bind(email.split("@")[0], finalUsername, email).run();

    profile = { id: result.meta.last_row_id as number, is_admin: 0 };
  }

  const { sessionId, maxAge } = await createSession(env, profile.id, !!profile.is_admin, true);

  return new Response(null, {
    status: 302,
    headers: {
      Location: data.returnTo || "/",
      "Set-Cookie": setSessionCookie(sessionId, maxAge),
    },
  });
};
```

```typescript
export interface FeedItem {
  title: string;
  link: string;
  description: string;
  pubDate: string;
  source: string;
}

export interface FeedSource {
  name: string;
  url: string;
  defaultCategory: string;
}

export const RSS_SOURCES: FeedSource[] = [
  { name: "NWP News", url: "https://www.nwpolytech.ca/news/rss", defaultCategory: "hiring" },
  { name: "City of GP", url: "https://www.cityofgp.com/feed", defaultCategory: "industry" },
  { name: "GP Chamber", url: "https://grandeprairiechamber.com/feed", defaultCategory: "industry" },
  { name: "Alberta Innovates", url: "https://albertainnovates.ca/feed", defaultCategory: "opportunity" },
  { name: "ATB Insights", url: "https://www.atb.com/company/insights/feed", defaultCategory: "industry" },
  { name: "Alberta Energy Regulator", url: "https://www.aer.ca/feed", defaultCategory: "industry" },
  { name: "GP Herald-Tribune", url: "https://www.dailyheraldtribune.com/feed", defaultCategory: "industry" },
];

export async function fetchFeedItems(source: FeedSource, sinceHours: number = 24): Promise<FeedItem[]> {
  try {
    const res = await fetch(source.url, {
      headers: { "User-Agent": "GrandePrairie.dev Intel Pipeline" },
    });
    if (!res.ok) return [];

    const text = await res.text();
    const items: FeedItem[] = [];
    const cutoff = Date.now() - sinceHours * 60 * 60 * 1000;

    // Simple XML parsing (RSS/Atom) — no library needed for basic feeds
    const itemMatches = text.matchAll(/<item>([\s\S]*?)<\/item>|<entry>([\s\S]*?)<\/entry>/gi);

    for (const match of itemMatches) {
      const block = match[1] ?? match[2] ?? "";

      const title = block.match(/<title[^>]*>([\s\S]*?)<\/title>/i)?.[1]?.replace(/<!\[CDATA\[([\s\S]*?)\]\]>/g, "$1").trim() ?? "";
      const link = block.match(/<link[^>]*href="([^"]+)"/i)?.[1] ?? block.match(/<link[^>]*>([\s\S]*?)<\/link>/i)?.[1]?.trim() ?? "";
      const description = block.match(/<description[^>]*>([\s\S]*?)<\/description>|<summary[^>]*>([\s\S]*?)<\/summary>|<content[^>]*>([\s\S]*?)<\/content>/i);
      const desc = (description?.[1] ?? description?.[2] ?? description?.[3] ?? "").replace(/<!\[CDATA\[([\s\S]*?)\]\]>/g, "$1").replace(/<[^>]+>/g, "").trim();
      const pubDateStr = block.match(/<pubDate>([\s\S]*?)<\/pubDate>|<published>([\s\S]*?)<\/published>|<updated>([\s\S]*?)<\/updated>/i);
      const pubDate = pubDateStr?.[1] ?? pubDateStr?.[2] ?? pubDateStr?.[3] ?? "";

      if (!title || !link) continue;

      // Filter by date if parseable
      if (pubDate) {
        const parsed = new Date(pubDate).getTime();
        if (!isNaN(parsed) && parsed < cutoff) continue;
      }

      items.push({ title, link, description: desc.slice(0, 500), pubDate, source: source.name });
    }

    return items;
  } catch {
    console.error(`RSS fetch failed for ${source.name}`);
    return [];
  }
}
```

import type { FeedItem } from "./rss";

export interface DraftedItem {
  title: string;
  body: string;
  category: string;
  tags: string[];
  relevant: boolean;
  sourceItem: FeedItem;
}

const SYSTEM_PROMPT = `You write Intel posts for GrandePrairie.dev, a tech community platform in Grande Prairie, Alberta (Peace Region).

For each news item, produce a JSON object:
{ "title": "...", "body": "2-3 sentence summary", "category": "hiring|industry|opportunity|events|project_activity", "tags": ["tag1", "tag2"], "relevant": true|false }

Only mark relevant=true if the item has a DIRECT connection to:
- Grande Prairie, Peace Region, or Northern Alberta (must name-drop the region or a regional entity)
- Technology, trades-tech, oil & gas tech, agri-tech, forestry tech IN the Peace Region
- Post-secondary education in the region (NWP, GPRC)
- Startup/business development specifically in the Peace Region
- Well licensing, drilling activity, or pipeline projects in the Montney/Peace Region

REJECT generic national/provincial news that merely mentions Alberta without a regional connection.
REJECT national AI/tech news that could apply anywhere. Be concise and practical — no hype.

Respond with a JSON array of objects. One per input item.`;

export async function draftIntelItems(apiKey: string, items: FeedItem[]): Promise<DraftedItem[]> {
  if (items.length === 0) return [];

  const userContent = items.map((item, i) =>
    `Item ${i + 1}:\nTitle: ${item.title}\nSource: ${item.source}\nDescription: ${item.description}\nLink: ${item.link}`
  ).join("\n\n");

  try {
    const res = await fetch("https://api.groq.com/openai/v1/chat/completions", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${apiKey}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: "llama-3.1-70b-versatile",
        messages: [
          { role: "system", content: SYSTEM_PROMPT },
          { role: "user", content: userContent },
        ],
        temperature: 0.3,
        response_format: { type: "json_object" },
      }),
    });

    const data = await res.json<{ choices: Array<{ message: { content: string } }> }>();
    const content = data.choices?.[0]?.message?.content ?? "[]";
    const parsed = JSON.parse(content);
    const results: any[] = Array.isArray(parsed) ? parsed : parsed.items ?? parsed.results ?? [];

    return results.map((r: any, i: number) => ({
      title: r.title ?? items[i]?.title ?? "",
      body: r.body ?? "",
      category: r.category ?? "industry",
      tags: Array.isArray(r.tags) ? r.tags : [],
      relevant: r.relevant === true,
      sourceItem: items[i],
    }));
  } catch (err) {
    console.error("Groq API failed:", err);
    return [];
  }
}

Title: ${title}
Body: ${body}
Category: ${category}
Source: ${sourceUrl}

Check:
1. Is the summary factually grounded in the source URL title/description?
2. Is the tone practical and non-hype?
3. Is the category correct?
4. Is the content relevant to Grande Prairie / Peace Region / Northern Alberta tech?

Respond with JSON only: { "verdict": "accept"|"reject"|"edit", "reason": "...", "edited_body": "..." }`,
        }],
      }),
    });

    console.error("Anthropic QC failed");
    return { verdict: "reject", reason: "QC API error" };
  }
}

### Task 9: Scheduled worker — the pipeline

**Files:**
- Create: `functions/_worker.ts` — scheduled event handler (or `src/pipeline.ts` if Pages doesn't support scheduled in functions/)

Note: Cloudflare Pages supports `_worker.ts` in the functions directory for advanced use cases including `scheduled` handlers. However, the standard approach for Pages is to put the scheduled handler in a separate worker or use `_worker.js` at the project root. For simplicity, we'll create a standalone pipeline script that can be run via `wrangler pages dev` or as a separate worker.

Actually, the simplest approach for Cloudflare Pages: add a `_worker.ts` export or use the Pages `onRequest` pattern with a secret-protected endpoint that the cron calls. Let's use a **protected API endpoint** approach — create `functions/api/pipeline/run.ts` that's admin-only (or secret-protected), and configure the cron to call it.

- [ ] **Step 1: Create pipeline runner endpoint**

`functions/api/pipeline/run.ts`:
```typescript
import type { Env } from "../../lib/env";
import { fetchFeedItems, RSS_SOURCES } from "../../lib/rss";
import { draftIntelItems } from "../../lib/groq";
import { qcIntelItem } from "../../lib/anthropic";
import { notifySlack } from "../../lib/slack";

export const onRequestPost: PagesFunction<Env> = async ({ request, env }) => {
  // Protect with a secret header (cron sends this)
  const authHeader = request.headers.get("X-Pipeline-Secret");
  const pipelineSecret = (env as any).PIPELINE_SECRET;
  if (!pipelineSecret || authHeader !== pipelineSecret) {
    return new Response("Forbidden", { status: 403 });
  }

  const startTime = Date.now();
  const runDate = new Date().toISOString().split("T")[0];

  // Step 1: Fetch RSS
  let allItems: Awaited<ReturnType<typeof fetchFeedItems>>[] = [];
  for (const source of RSS_SOURCES) {
    const items = await fetchFeedItems(source);
    allItems.push(...items.map(item => ({ ...item })));
  }

  // Deduplicate against existing intel
  const deduped = [];
  for (const item of allItems) {
    const existing = await env.DB.prepare(
      "SELECT id FROM intel WHERE source_url = ? OR title = ?",
    ).bind(item.link, item.title).first();
    if (!existing) deduped.push(item);
  }

  // Step 2: Draft with Groq
  const groqKey = (env as any).GROQ_API_KEY as string;
  const drafted = groqKey ? await draftIntelItems(groqKey, deduped) : [];
  const relevant = drafted.filter(d => d.relevant);

  // Step 3: QC with Anthropic
  const anthropicKey = (env as any).ANTHROPIC_API_KEY as string;
  const accepted: typeof relevant = [];
  const rejectReasons: string[] = [];

  for (const item of relevant) {
    if (!anthropicKey) {
      accepted.push(item); // Skip QC if no key
      continue;
    }
    const qc = await qcIntelItem(anthropicKey, item.title, item.body, item.category, item.sourceItem.link);
    if (qc.verdict === "accept") {
      accepted.push(item);
    } else if (qc.verdict === "edit" && qc.editedBody) {
      accepted.push({ ...item, body: qc.editedBody });
    } else {
      rejectReasons.push(`${item.title}: ${qc.reason}`);
    }
  }

  // Step 4: Publish
  for (const item of accepted) {
    await env.DB.prepare(
      `INSERT INTO intel (title, body, category, source_url, tags, is_automated, source_feed, author_id)
       VALUES (?, ?, ?, ?, ?, 1, ?, NULL)`,
    ).bind(
      item.title, item.body, item.category, item.sourceItem.link,
      JSON.stringify(item.tags), item.sourceItem.source,
    ).run();
  }

  // Log pipeline run
  const duration = Date.now() - startTime;
  await env.DB.prepare(
    `INSERT INTO pipeline_runs (run_date, sources_checked, items_fetched, items_drafted, items_accepted, items_rejected, reject_reasons, duration_ms)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
  ).bind(
    runDate, RSS_SOURCES.length, allItems.length, relevant.length,
    accepted.length, rejectReasons.length, JSON.stringify(rejectReasons), duration,
  ).run();

  // Step 5: Slack
  const categoryBreakdown = accepted.reduce((acc, item) => {
    acc[item.category] = (acc[item.category] ?? 0) + 1;
    return acc;
  }, {} as Record<string, number>);
  const breakdown = Object.entries(categoryBreakdown).map(([k, v]) => `${v} ${k}`).join(", ");

  if (accepted.length > 0) {
    await notifySlack(env, `📰 Daily Intel — ${runDate}\nPublished: ${accepted.length} posts (${breakdown})\nRejected: ${rejectReasons.length} items\nSources: ${RSS_SOURCES.length} feeds\nDuration: ${duration}ms`);
  } else {
    await notifySlack(env, `📰 Daily Intel — ${runDate}\nNo relevant items found today. ${allItems.length} items checked from ${RSS_SOURCES.length} feeds.`);
  }

  return Response.json({
    run_date: runDate,
    sources: RSS_SOURCES.length,
    fetched: allItems.length,
    deduped: deduped.length,
    drafted: relevant.length,
    accepted: accepted.length,
    rejected: rejectReasons.length,
    duration_ms: duration,
  });
};
```
